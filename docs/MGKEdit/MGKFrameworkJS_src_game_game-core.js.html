                if (this.reInit === true) {
                    preState = this.preInitFunc(this);
                }
                if (preState === true) {
                    this.comMan.Init();
                    if (!this.gameLogic.VInit()) {
                        return false;
                    }
                    if (this.reInit === true) {
                        this.reInit = false;
                        this.isInitialized = true;
                        this.VUpdate();
                    }
                }
                else {
                    this.reInit = false;
                    this.Quit();
                    return false;
                }
            }
            this.isInitialized = true;
            if (this.reInit === true) {
                this.reInit = false;

            return true;

                return;
            }

                return;

                    events.returnValue = false;
                    events.preventDefault();

            };
        }

        }
    }
